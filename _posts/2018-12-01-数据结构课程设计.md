## 题目

### 校园导游程序及通信线路设计

设计校园平面图，所含景点不少于10个。以图中顶点表示校内各景点，存放景点名称、代号、简介等信息；以边表示路径，存放路径长度等相关信息。

1. 为来访客人提供图中任意景点相关信息的查询。

2. 为来访客人提供图中任意2个景点的问路查询,即查询任意两个景点之间的一条最短的简单路径。

3. 以尽可能低的造价建造景点间的通信网络把这些景点联系在一起，每条通信线路的造价与景点间的距离成正比。

## 开发工具

Code::Blocks + gcc + gdb

## 开发语言

C++

## 界面

CLI(command-line interface)

## 算法设计

- 从文件中读取数据建立图，邻接矩阵存储图中边的权值，用vector顺序存储点的信息；

- 用STL中的unordered_map实现名称与代号（vector数组下标）的映射，便于查找；

- 用Floyd算法求任意两点最短路径；

- 用Prim算法求图的最小生成树。

```cpp
#include <unordered_map>
#include <iostream>
#include <string>
#include <vector>
#include <cfloat>
#define START 0
#define END(n) (n)
#define ERROR -1
#define NONE -1
#define COLLECTED 0
#define MAX_V_NUM 32
using namespace std;
#ifndef MAP
#define MAP
typedef unordered_map<string, int> Map;
#endif
#ifndef EDGE
#define EDGE
typedef struct _edge
{
    string v1, v2;

#endif

    void PrintMST();
private:
    void PrintInBetweenVertex(int src, int dst);
    void Floyd();
    double Prim();
    int FindMinDist(double dist[]);
    int numVertex, numEdge;
    double graph[MAX_V_NUM][MAX_V_NUM];
    double dist[MAX_V_NUM][MAX_V_NUM];
```

```cpp
//graph.cpp
#include "graph.h"

Graph::Graph(int n) : numVertex(0), numEdge(0)
{
    for (int i = START; i < END(n); ++i)
        for (int j = START; j < END(n); ++j)
            graph[i][j] = DBL_MAX;
}
void Graph::InsertVertex(Vertex &V)
{
    if (map.find(V.name) == map.end())
    {
        data.push_back(V);
        map[V.name] = numVertex++;
    }
    else
		cout << "点的信息与已有的重复"<<endl;
}
void Graph::InsertEdge(Edge &E)
{
    if (map.find(E.v1) != map.end() && map.find(E.v2) != map.end())
    {
        int v1 = map[E.v1], v2 = map[E.v2];
        graph[v1][v2] = E.length;
        graph[v2][v1] = E.length;
        ++numEdge;
    }
    else
        cout << "边的信息错误" << endl;
}
void Graph::DisplayAll()
{
    for (int i = 0; i < numVertex; ++i)
        cout << data[i].name << " ";
    cout << endl;
}
void Graph::DisplayData(string &name)
{
    if (map.find(name) != map.end())
    {
        int i = map[name];
        cout << "代号：" << i << endl;
        cout << "名称: " << data[i].name << endl;
        cout << "描述: " << data[i].desc << endl;
    }
    else
        cout << "找不到\"" << name << "\"该景点。" << endl;
}
void Graph::PrintShortestPath(string& src, string& dst)
{
    if (map.find(src) != map.end() && map.find(dst) != map.end())
    {
        Floyd();
        cout << src << "->";
        PrintInBetweenVertex(map[src], map[dst]);
        cout << dst << endl;
    }
    else
        cout << "出发地或目的地名称输入错误。" << endl;
}
double Graph::Prim()
{
    double dist[MAX_V_NUM], weightSum = 0;
    int parent[MAX_V_NUM], V, W;
    for (V = START; V < END(numVertex); ++V)
    {
        dist[V] = graph[START][V];
        parent[V] = START;
    }
    dist[START] = COLLECTED;
    parent[START] = NONE;
    while ((V = FindMinDist(dist)) != ERROR)
    {
        MST.push_back({data[parent[V]].name, data[V].name, dist[V]});
        weightSum += dist[V];
        dist[V] = COLLECTED;
        for (W = START; W < END(numVertex); ++W)
            if (dist[W] != COLLECTED && graph[V][W] < DBL_MAX)
            {
                if (graph[V][W] < dist[W])
                {
                    dist[W] = graph[V][W];
                    parent[W] = V;
                }
            }
    }
    return weightSum;
}
void Graph::PrintMST()
{
    cout << "最低造价：" << Prim() << endl;
    for (int i = 0; i < MST.size(); ++i)
		cout << MST[i].v1 << "---" << MST[i].length << "---" << MST[i].v2 << endl;
}

void Graph::PrintInBetweenVertex(int src, int dst)
{
    if (path[src][dst] == -1)
        return;
    PrintInBetweenVertex(src, path[src][dst]);
    cout << data[path[src][dst]].name << "->";
    PrintInBetweenVertex(path[src][dst], dst);
}
void Graph::Floyd()
{
    for (int i = 0; i < numVertex; ++i)
        for (int j = 0; j < numVertex; ++j)
        {
            dist[i][j] = DBL_MAX;
            path[i][j] = -1;
        }
    for (int i = 0; i < numVertex; ++i)
        for (int j = 0; j < numVertex; ++j)
            dist[i][j] = graph[i][j];
    for (int k = 0; k < numVertex; ++k)
        for (int i = 0; i < numVertex; ++i)
            for (int j = 0; j < numVertex; ++j)
                if (dist[i][k] + dist[k][j] < dist[i][j])
                {
                    dist[i][j] = dist[i][k] + dist[k][j];
                    path[i][j] = k;
                }
}
int Graph::FindMinDist(double dist[])
{
    int minV, V;
    double minDist = DBL_MAX;
    for (V = START; V < END(numVertex); ++V)
        if (dist[V] != COLLECTED && dist[V] < minDist)
        {
            minDist = dist[V];
            minV = V;
        }
    if (minDist < DBL_MAX)
        return minV;
    else
        return ERROR;
}
```
