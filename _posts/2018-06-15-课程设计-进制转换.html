题目

五、八、十、十二、十六进制数之间相互转换。

进制转换方法

  1. 十进制数表示成r进制数

    设十进制数d.c，对应r进制数为an, an-1…a1a0.a-1a-2…a-m，则有

    d.c=anrn+ an-1rn-1+ …+a1r1+ a0r0+a-1r-1+ a-2 r-2+ …+a-m r-m，其中:

    d = anrn+ an-1rn-1+ …+a1r1+ a0r0，0.c = a-1r-1+ a-2 r-2+ …+a-m r-m

  2. r进制数为anan-1…a1a0.a-1a-2…a-m，转换到十进制数按公式计算即可。

  3. 十进制数d.c，转换到r进制数

    设对应的r进制数为anan-1…a1a0.a-1a-2…a-m，则可按整数部分和小数部分分别转换：

    (1) 整数部分转换

    a0=d % r, d1=d / r, d1=anrn-1+ an-1rn-2+ …+a1

    a1=d1% r, d2=d1 / r, d2=anrn-2+ an-1rn-3+ …+a2

    据此,可依次求出: a0 , a1 , … , an-1 , an；

    (2) 小数部分转换

    0.c * r = a-1.c1，0.c1 * r = a-2.c2，…据此,可依次求出: a-1 , a-2, … , a-m。

进制转换(附带输出结果)

思路：

  1. r1进制数转换到10进制数；
  2. 10进制数转换到r2进制数；
  3. 整数与小数部分分别转换，边处理边输出；

注：可能运算过程中有精度损失

xxxxxxxxxx
void HexConvert(char s[], int len, int r1, int r2)
{
    if(s[0] == '-') //负数当正数处理， 先输出负号
    {
        putchar('-');
        s[0] = '0';
    }
    double dec = 0.0;
    char *p;
    int dotPos = ((p = strchr(s, '.')) != NULL) ? p - &s[0] : len; //以小数点为界
    for (int i = dotPos - 1, j = 0; i >= 0; --i, ++j) //整数部分
        dec += asciiMap[s[i]] * pow(r1, j);
    if (p != NULL)
    for (int i = dotPos + 1, j = -1; s[i] != '\0'; ++i, --j) //小数部分
        dec += asciiMap[s[i]] * pow(r1, j);
    if (r2 == 10)
    {
        printf("%g\n", dec);
        return;
    }
    int intPart = (int)dec;
    double floatPart = dec - intPart;
    int tmp[100];
    if (intPart) //如果整数部分不为零
    {
        int i;
        for (i = 0; intPart; ++i, intPart /= r2) //模拟短除法
            tmp[i] = intPart % r2;
        while (i--)
            putchar(numMap[tmp[i]]);
    }
    else
        putchar('0');
    if (floatPart >= EPS) //如果存在小数部分
    {
        putchar('.');
        for(int i = 0; i < PRECISION; ++i)
        {
            floatPart *= r2;
            tmp[i] = (int)floatPart;
            floatPart -= tmp[i];
            putchar(numMap[tmp[i]]);
        }
        putchar('\n');
    }
    else
        putchar('\n');
}

                {
                case INVALID:
                    puts("输入不合法！请重新输入：");
                    Clear();
                    jump = 0;
                    break;
                case VALID:
                    HexConvert(s, len, r1, r2);
                    break;
                default:
                    break;
                }
                if(jump)
                    break;
            }
        }
        system("pause");
        system("cls");
    }
    return 0;
}
